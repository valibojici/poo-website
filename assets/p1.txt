#BEGIN_PROBLEM
#include <iostream>

using namespace std;

class Vector2D{
private:
    int x,y;
public:
    Vector2D(int x, int y) : x(x), y(y) {}
    void afis(){
        cout << '(' << x << ',' << y << ')';
    }
};

int main(){
    Vector2D a(1,2), b(2,3), c;
    a.afis();
}
#END_PROBLEM

#BEGIN_SOLUTION
<div>
Programul nu compileaza, avem eroare la linia 16: <inline>Vector2D a(1,2), b(2,3), c;</inline>
</div>
<error>error: no matching function for call to 'Vector2D::Vector2D()' </error>
<div>
Pentru ca nu avem contructor cu parametri impliciti nu se poate crea o variabila de tip Vector2D fara sa specifici valorile parametrilor din constructor (e vorba despre variabila <var>c</var>)
</div>
<div>
Pentru ca programul sa mearga sunt mai multe variante: 
</div>
<div>
1) stergi variabila <var>c</var> de la linia 16 care devine <inline>Vector2D a(1,2), b(2,3);</inline>
</div>
<div>
2) adaugi valori implicite pentru contructor, adica linia 9 sa devina <inline>Vector2D(int x = 0, int y = 0) : x(x), y(y) {}</inline>. Poti pune si alte valori in afara de 0, nu conteaza
</div>
<div>
3) adaugi constructor fara parametri:
<block>
#include <iostream>

using namespace std;

class Vector2D{
private:
    int x,y;
public:

    Vector2D() {} // constructor fara parametrii

    Vector2D(int x, int y) : x(x), y(y) {}
    void afis(){
        cout << '(' << x << ',' << y << ')';
    }
};

int main(){
    Vector2D a(1,2), b(2,3), c;
    a.afis();
}
</block>
</div>
#END_SOLUTION
