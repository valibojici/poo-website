#BEGIN_PROBLEM
#include <iostream>
using namespace std;

int myFunction(int A = 1, int B = 2, int C) {
	int temp = A;
	for (int i = 0; i < B; ++i)
	{
		temp += C;
		C--;
	}
	return temp;
}

int main() {
	int num = myFunction(1, 2, 3);
	cout << num << endl;
}
#END_PROBLEM


#BEGIN_SOLUTION

</error>

</div>
